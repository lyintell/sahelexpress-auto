# DB SQL pour Supabase

Copiez-collez ce script dans l'editeur SQL de Supabase.

```sql
begin;

create extension if not exists pgcrypto;

create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = timezone('utc', now());
  return new;
end;
$$;

create table if not exists public.users (
  id bigint generated by default as identity primary key,
  nom text not null,
  tel text not null,
  email text not null unique,
  type_utilisateur text not null check (type_utilisateur in ('Admin', 'Editeur')),
  mot_de_passe_hash text,
  created_at timestamptz not null default timezone('utc', now()),
  updated_at timestamptz not null default timezone('utc', now())
);

create table if not exists public.marques (
  id bigint generated by default as identity primary key,
  nom text not null unique,
  created_at timestamptz not null default timezone('utc', now()),
  updated_at timestamptz not null default timezone('utc', now())
);

create table if not exists public.modeles (
  id bigint generated by default as identity primary key,
  nom text not null,
  marque_id bigint not null references public.marques(id) on delete cascade,
  created_at timestamptz not null default timezone('utc', now()),
  updated_at timestamptz not null default timezone('utc', now()),
  constraint modeles_nom_marque_unique unique (nom, marque_id)
);

create table if not exists public.vehicules (
  id bigint generated by default as identity primary key,
  nom text not null,
  model_id bigint not null references public.modeles(id) on delete cascade,
  marque_id bigint not null references public.marques(id) on delete cascade,
  marque_nom text,
  model_nom text,
  annee integer not null check (annee between 1900 and 2100),
  couleur text not null,
  type_carb text not null check (type_carb in ('essence', 'gasoil')),
  type_transm text not null check (type_transm in ('auto', 'manuel')),
  type_moteur text not null check (type_moteur in ('8 cyl', '6 cyl', '4 cyl', 'hybrid')),
  kilometrage integer,
  prix_vente numeric(12, 2),
  prix_achat numeric(12, 2),
  montant_transport numeric(12, 2),
  montant_dedouanement numeric(12, 2),
  description text,
  ind_en_vedette boolean not null default true,
  ind_etat text not null check (ind_etat in ('vendu', 'disponible', 'non-disponible')),
  ajoute_par bigint not null references public.users(id),
  ajoute_le timestamptz not null default timezone('utc', now()),
  modifie_par bigint references public.users(id),
  modifie_le timestamptz,
  image text not null,
  image_2 text,
  image_3 text,
  image_4 text,
  image_5 text,
  image_6 text,
  image_7 text,
  image_8 text,
  image_9 text,
  image_10 text,
  created_at timestamptz not null default timezone('utc', now()),
  updated_at timestamptz not null default timezone('utc', now())
);

alter table public.vehicules add column if not exists prix_vente numeric(12, 2);
alter table public.vehicules add column if not exists prix_achat numeric(12, 2);
alter table public.vehicules add column if not exists montant_transport numeric(12, 2);
alter table public.vehicules add column if not exists montant_dedouanement numeric(12, 2);

create table if not exists public.contact_details (
  id smallint primary key default 1,
  phone text not null,
  email text not null,
  location text not null,
  created_at timestamptz not null default timezone('utc', now()),
  updated_at timestamptz not null default timezone('utc', now()),
  constraint contact_details_single_row check (id = 1)
);

create table if not exists public.hero_content (
  id smallint primary key default 1,
  title text not null,
  image text not null,
  created_at timestamptz not null default timezone('utc', now()),
  updated_at timestamptz not null default timezone('utc', now()),
  constraint hero_content_single_row check (id = 1)
);

create index if not exists idx_modeles_marque_id on public.modeles(marque_id);
create index if not exists idx_vehicules_model_id on public.vehicules(model_id);
create index if not exists idx_vehicules_marque_id on public.vehicules(marque_id);
create index if not exists idx_vehicules_etat on public.vehicules(ind_etat);
create index if not exists idx_vehicules_vedette on public.vehicules(ind_en_vedette);

create or replace trigger users_set_updated_at
before update on public.users
for each row execute function public.set_updated_at();

create or replace trigger marques_set_updated_at
before update on public.marques
for each row execute function public.set_updated_at();

create or replace trigger modeles_set_updated_at
before update on public.modeles
for each row execute function public.set_updated_at();

create or replace trigger vehicules_set_updated_at
before update on public.vehicules
for each row execute function public.set_updated_at();

create or replace trigger contact_details_set_updated_at
before update on public.contact_details
for each row execute function public.set_updated_at();

create or replace trigger hero_content_set_updated_at
before update on public.hero_content
for each row execute function public.set_updated_at();

alter table public.users enable row level security;
alter table public.marques enable row level security;
alter table public.modeles enable row level security;
alter table public.vehicules enable row level security;
alter table public.contact_details enable row level security;
alter table public.hero_content enable row level security;

drop policy if exists "public read marques" on public.marques;
create policy "public read marques" on public.marques
for select using (true);

drop policy if exists "public read modeles" on public.modeles;
create policy "public read modeles" on public.modeles
for select using (true);

drop policy if exists "public read vehicules" on public.vehicules;
create policy "public read vehicules" on public.vehicules
for select using (true);

drop policy if exists "public read contact" on public.contact_details;
create policy "public read contact" on public.contact_details
for select using (true);

drop policy if exists "public read hero" on public.hero_content;
create policy "public read hero" on public.hero_content
for select using (true);

drop policy if exists "authenticated read users" on public.users;
create policy "authenticated read users" on public.users
for select to authenticated using (true);

drop policy if exists "service role manage users" on public.users;
create policy "service role manage users" on public.users
for all to service_role using (true) with check (true);
drop policy if exists "service role manage marques" on public.marques;
create policy "service role manage marques" on public.marques
for all to service_role using (true) with check (true);
drop policy if exists "service role manage modeles" on public.modeles;
create policy "service role manage modeles" on public.modeles
for all to service_role using (true) with check (true);
drop policy if exists "service role manage vehicules" on public.vehicules;
create policy "service role manage vehicules" on public.vehicules
for all to service_role using (true) with check (true);
drop policy if exists "service role manage contact" on public.contact_details;
create policy "service role manage contact" on public.contact_details
for all to service_role using (true) with check (true);
drop policy if exists "service role manage hero" on public.hero_content;
create policy "service role manage hero" on public.hero_content
for all to service_role using (true) with check (true);

insert into public.contact_details (id, phone, email, location)
values (1, '+223 00 00 00 00', 'contact@sahelexpress-auto.com', 'Bamako, Mali')
on conflict (id) do nothing;

insert into public.hero_content (id, title, image)
values (1, 'Sahel Express Auto', '')
on conflict (id) do nothing;

commit;
```

## Notes

- `mot_de_passe_hash` est prevu pour eviter le stockage du mot de passe en clair. Si vous utilisez Supabase Auth, gardez cette colonne optionnelle ou retirez-la plus tard.
- `marque_nom` et `model_nom` sont gardes pour coller a l'etat actuel du frontend, meme si ces valeurs peuvent etre reconstruites par jointure.
- Les colonnes `prix_vente`, `prix_achat`, `montant_transport` et `montant_dedouanement` sont optionnelles pour permettre une migration progressive du frontend.
- Les suppressions en cascade sont actives entre `marques -> modeles -> vehicules` et `modeles -> vehicules`.
